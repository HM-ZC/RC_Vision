233 UKF()
234 : alpha_(1e-3), beta_(2.0), kappa_(0.0) {
235 // 计算权重
236 double lambda = std::pow(alpha_, 2) * (StateDim + kappa_) - StateDim;
237 weights_.resize(2 * StateDim + 1);
238 weights_(0) = lambda / (StateDim + lambda);
239 for(int i = 1; i < 2 * StateDim + 1; ++i) {
240 weights_(i) = 1 / (2 * (StateDim + lambda));
241 }
242 }

286 void predict() {
287 // 生成 sigma points
288 Eigen::Matrix<T, StateDim, 2 * StateDim + 1> sigma_points = computeSigmaPoints(KalmanFilter<T, StateDim, MeasurementDim>::x_, KalmanFilter<T, StateDim, MeasurementDim>::P_);
289
290 // 通过状态转移函数传递 sigma 点
291 Eigen::Matrix<T, StateDim, 2 * StateDim + 1> sigma_points_pred;
292 for(int i = 0; i < 2 * StateDim + 1; ++i) {
293 sigma_points_pred.col(i) = f_(sigma_points.col(i));
294 }
295
296 // 计算预测状态均值
297 Eigen::Matrix<T, StateDim, 1> x_pred = Eigen::Matrix<T, StateDim, 1>::Zero();
298 for(int i = 0; i < 2 * StateDim + 1; ++i) {
299 x_pred += weights_(i) * sigma_points_pred.col(i);
300 }
301
302 // 计算预测状态协方差
303 Eigen::Matrix<T, StateDim, StateDim> P_pred = Eigen::Matrix<T, StateDim, StateDim>::Zero();
304 for(int i = 0; i < 2 * StateDim + 1; ++i) {
305 Eigen::Matrix<T, StateDim, 1> diff = sigma_points_pred.col(i) - x_pred;
306 P_pred += weights_(i) * diff * diff.transpose();
307 }
308 P_pred += KalmanFilter<T, StateDim, MeasurementDim>::Q_;
309
310 // 更新状态和协方差
311 KalmanFilter<T, StateDim, MeasurementDim>::x_ = x_pred;
312 KalmanFilter<T, StateDim, MeasurementDim>::P_ = P_pred;
313 }

322 void update(const Eigen::Matrix<T, MeasurementDim, 1>& z) {
323 // 生成 sigma points from predicted state
324 Eigen::Matrix<T, StateDim, 2 * StateDim + 1> sigma_points = computeSigmaPoints(KalmanFilter<T, StateDim, MeasurementDim>::x_, KalmanFilter<T, StateDim, MeasurementDim>::P_);
325
326 // 通过测量函数传递 sigma 点
327 Eigen::Matrix<T, MeasurementDim, 2 * StateDim + 1> Z_sigma;
328 for(int i = 0; i < 2 * StateDim + 1; ++i) {
329 Z_sigma.col(i) = h_(sigma_points.col(i));
330 }
331
332 // 计算预测测量均值
333 Eigen::Matrix<T, MeasurementDim, 1> z_pred = Eigen::Matrix<T, MeasurementDim, 1>::Zero();
334 for(int i = 0; i < 2 * StateDim + 1; ++i) {
335 z_pred += weights_(i) * Z_sigma.col(i);
336 }
337
338 // 计算预测测量协方差
339 Eigen::Matrix<T, MeasurementDim, MeasurementDim> S = Eigen::Matrix<T, MeasurementDim, MeasurementDim>::Zero();
340 for(int i = 0; i < 2 * StateDim + 1; ++i) {
341 Eigen::Matrix<T, MeasurementDim, 1> diff = Z_sigma.col(i) - z_pred;
342 S += weights_(i) * diff * diff.transpose();
343 }
344 S += KalmanFilter<T, StateDim, MeasurementDim>::R_;
345
346 // 计算交叉协方差
347 Eigen::Matrix<T, StateDim, MeasurementDim> Tc = Eigen::Matrix<T, StateDim, MeasurementDim>::Zero();
348 for(int i = 0; i < 2 * StateDim + 1; ++i) {
349 Eigen::Matrix<T, StateDim, 1> x_diff = sigma_points.col(i) - KalmanFilter<T, StateDim, MeasurementDim>::x_;
350 Eigen::Matrix<T, MeasurementDim, 1> z_diff = Z_sigma.col(i) - z_pred;
351 Tc += weights_(i) * x_diff * z_diff.transpose();
352 }
353
354 // 计算卡尔曼增益
355 Eigen::Matrix<T, StateDim, MeasurementDim> K = Tc * S.inverse();
356
357 // 更新状态均值和协方差矩阵
358 KalmanFilter<T, StateDim, MeasurementDim>::x_ += K * (z - z_pred);
359 KalmanFilter<T, StateDim, MeasurementDim>::P_ -= K * S * K.transpose();
360 }
361
362 private:
363 // 状态转移函数
364 std::function<Eigen::Matrix<T, StateDim, 1>(const Eigen::Matrix<T, StateDim, 1>&)> f_;
365
366 // 测量函数
367 std::function<Eigen::Matrix<T, MeasurementDim, 1>(const Eigen::Matrix<T, StateDim, 1>&)> h_;
368
369 // 状态转移函数的雅可比矩阵函数
370 std::function<Eigen::Matrix<T, StateDim, StateDim>(const Eigen::Matrix<T, StateDim, 1>&)> F_jacobian_;
371
372 // 测量函数的雅可比矩阵函数
373 std::function<Eigen::Matrix<T, MeasurementDim, StateDim>(const Eigen::Matrix<T, StateDim, 1>&)> H_jacobian_;
374
375 // 权重向量
376 Eigen::Matrix<T, 2 * StateDim + 1, 1> weights_;
377
378 // 无迹卡尔曼滤波器参数
379 double alpha_;
380 double beta_;
381 double kappa_;
390 Eigen::Matrix<T, StateDim, 2 * StateDim + 1> computeSigmaPoints(
391 const Eigen::Matrix<T, StateDim, 1>& x,
392 const Eigen::Matrix<T, StateDim, StateDim>& P) const {
393
394 Eigen::Matrix<T, StateDim, 2 * StateDim + 1> sigma_points;
395 Eigen::Matrix<T, StateDim, StateDim> sqrt_P = P.llt().matrixL();
396
397 sigma_points.col(0) = x;
398 double lambda = std::pow(alpha_, 2) * (StateDim + 0.0) - StateDim; // kappa_ = 0
399
400 for(int i = 0; i < StateDim; ++i) {
401 sigma_points.col(i + 1) = x + std::sqrt(StateDim + lambda) * sqrt_P.col(i);
402 sigma_points.col(i + 1 + StateDim) = x - std::sqrt(StateDim + lambda) * sqrt_P.col(i);
403 }
404
405 return sigma_points;
406 }

417 template <typename T, int StateDim, int MeasurementDim>
418 class ParticleFilter {
419 public:
425 ParticleFilter(int num_particles)
426 : num_particles_(num_particles), gen_(rd_()), dist_(0.0, 1.0) {
427 particles_.resize(num_particles_);
428 weights_.resize(num_particles_, 1.0 / num_particles_);
429 }
430
437 void init(const Eigen::Matrix<T, StateDim, 1>& initial_state,
438 const Eigen::Matrix<T, StateDim, StateDim>& initial_covariance) {
439 Eigen::Matrix<T, StateDim, StateDim> cov_sqrt = initial_covariance.llt().matrixL();
440 for(int i = 0; i < num_particles_; ++i) {
441 particles_[i] = initial_state + cov_sqrt * Eigen::Matrix<T, StateDim, 1>::Random();
442 }
443 }
444
453 void predict(const std::function<Eigen::Matrix<T, StateDim, 1>(const Eigen::Matrix<T, StateDim, 1>&)>& f,
454 const Eigen::Matrix<T, StateDim, StateDim>& Q) {
455 Eigen::Matrix<T, StateDim, StateDim> cov_sqrt = Q.llt().matrixL();
456 for(int i = 0; i < num_particles_; ++i) {
457 particles_[i] = f(particles_[i]) + cov_sqrt * Eigen::Matrix<T, StateDim, 1>::Random();
458 }
459 }
460
470 void update(const Eigen::Matrix<T, MeasurementDim, 1>& z,
471 const std::function<Eigen::Matrix<T, MeasurementDim, 1>(const Eigen::Matrix<T, StateDim, 1>&)>& h,
472 const Eigen::Matrix<T, MeasurementDim, MeasurementDim>& R) {
473 for(int i = 0; i < num_particles_; ++i) {
474 Eigen::Matrix<T, MeasurementDim, 1> z_pred = h(particles_[i]);
475 Eigen::Matrix<T, MeasurementDim, MeasurementDim> S = R;
476 Eigen::Matrix<T, MeasurementDim, 1> y = z - z_pred;
477 // 假设高斯噪声，计算似然度
478 T exponent = -0.5 * y.transpose() * S.inverse() * y;
479 T likelihood = std::exp(exponent);
480 weights_[i] *= likelihood;
481 }
482 normalizeWeights();
483 resample();
484 }
485
491 Eigen::Matrix<T, StateDim, 1> getState() const {
492 Eigen::Matrix<T, StateDim, 1> state = Eigen::Matrix<T, StateDim, 1>::Zero();
493 for(int i = 0; i < num_particles_; ++i) {
494 state += weights_[i] * particles_[i];
495 }
496 return state;
497 }
498
499 private:
500 int num_particles_;
501 std::vector<Eigen::Matrix<T, StateDim, 1>> particles_;
502 std::vector<T> weights_;
503 std::random_device rd_;
504 std::mt19937 gen_;
505 std::uniform_real_distribution<> dist_;
512 void normalizeWeights() {
513 T sum = 0.0;
514 for(auto w : weights_) sum += w;
515 for(auto& w : weights_) w /= sum;
516 }
517
523 void resample() {
524 std::vector<Eigen::Matrix<T, StateDim, 1>> new_particles;
525 new_particles.reserve(num_particles_);
526
527 // 创建累积分布
528 std::vector<T> cumulative_weights(num_particles_, 0.0);
529 cumulative_weights[0] = weights_[0];
530 for(int i = 1; i < num_particles_; ++i) {
531 cumulative_weights[i] = cumulative_weights[i-1] + weights_[i];
532 }
533
534 // 重采样
535 for(int i = 0; i < num_particles_; ++i) {
536 T rand_weight = dist_(gen_);
537 auto it = std::lower_bound(cumulative_weights.begin(), cumulative_weights.end(), rand_weight);
538 int index = std::distance(cumulative_weights.begin(), it);
539 new_particles.push_back(particles_[index]);
540 }
541
542 particles_ = std::move(new_particles);
543 std::fill(weights_.begin(), weights_.end(), 1.0 / num_particles_);
544 }
545 };
