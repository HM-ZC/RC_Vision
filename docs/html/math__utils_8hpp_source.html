74 double MathUtils::closestPointsBetweenSegments(const Eigen::Vector3d& p1, const Eigen::Vector3d& p2,
75 const Eigen::Vector3d& p3, const Eigen::Vector3d& p4,
76 Eigen::Vector3d& closestPoint1, Eigen::Vector3d& closestPoint2) {
77 Eigen::Vector3d u = p2 - p1;
78 Eigen::Vector3d v = p4 - p3;
79 Eigen::Vector3d w = p1 - p3;
80 double a = u.dot(u); // always >= 0
81 double b = u.dot(v);
82 double c = v.dot(v); // always >= 0
83 double d = u.dot(w);
84 double e = v.dot(w);
85 double D = a*c - b*b; // always >= 0
86 double sc, tc;
87
88 // compute the line parameters of the two closest points
89 if (D < 1e-8) { // the lines are almost parallel
90 sc = 0.0;
91 tc = (b > c ? d/b : e/c); // use the largest denominator
92 }
93 else {
94 sc = (b*e - c*d) / D;
95 tc = (a*e - b*d) / D;
96 }
97
98 // clamp sc to [0,1] to stay within the segments
99 sc = std::min(std::max(sc, 0.0), 1.0);
100 tc = std::min(std::max(tc, 0.0), 1.0);
101
102 closestPoint1 = p1 + sc * u;
103 closestPoint2 = p3 + tc * v;
104 return (closestPoint1 - closestPoint2).norm();
105 }

133 bool MathUtils::fitPlane(const std::vector<Eigen::Vector3d>& points,
134 Eigen::Vector3d& plane_point,
135 Eigen::Vector3d& plane_normal) {
136 if (points.size() < 3) {
137 Logger::getInstance().log(Logger::LogLevel::ERROR, "Not enough points to fit a plane.");
138 return false;
139 }
140
141 // Compute the centroid
142 Eigen::Vector3d centroid(0, 0, 0);
143 for(const auto& p : points) {
144 centroid += p;
145 }
146 centroid /= points.size();
147
148 // Compute the covariance matrix
149 Eigen::Matrix3d cov = Eigen::Matrix3d::Zero();
150 for(const auto& p : points) {
151 Eigen::Vector3d centered = p - centroid;
152 cov += centered * centered.transpose();
153 }
154
155 // Perform Eigen decomposition
156 Eigen::SelfAdjointEigenSolver<Eigen::Matrix3d> solver(cov);
157 if(solver.info() != Eigen::Success) {
158 Logger::getInstance().log(Logger::LogLevel::ERROR, "Eigen decomposition failed.");
159 return false;
160 }
161
162 // The normal of the plane is the eigenvector corresponding to the smallest eigenvalue
163 plane_normal = solver.eigenvectors().col(0).normalized();
164 plane_point = centroid;
165
166 return true;
167 }
